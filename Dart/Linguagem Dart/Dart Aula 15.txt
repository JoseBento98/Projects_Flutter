Reescrita de Métodos
posso puchar o metodo para dentro do objeto especifico
e alterar seus comandos
( Atravez do @override e String toString )
inserido dentro do objeto que ira puchar o codigo
do método.
toda classe no dart extende um object




class Animal{

	String nome;
	double peso;

	Animal(this.nome,this.peso);

	void correr () {
	print("$nome Correu");	
	}

	void FazerSom() {
	print("$nome fez algum som!");
	}



}

classe cachorro extends Animal {

	int fofura;	

	cachorro(String nome, double peso, this.fofura) :  super(nome,peso);

	void brincar() {
	fofura += 10;
	print("Fofura do $nome aumentou para $fofura!!");
	}

	@override
	void FazerSom() {
	print("$nome fez Au AU!");
	}

	@override
	String toString() {
		return "Cachorro | nome: $nome, Peso: $peso, Fofura: $fofura";
	}
}
